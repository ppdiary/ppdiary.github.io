  ```css
/* 
   =========================================
   左側鋼琴裝飾
   =========================================
*/

.fixed-piano-container {
  display: none;
}

@media (min-width: 768px) {
  .fixed-piano-container {
  display: block; /* 讓容器在大螢幕時復活 */

  position: fixed;
  top: 0;
  bottom: 0;
  left: 0;         
  width: 21px;    
  
  background-image:
    /* --- 第一層：Bb0 單獨黑鍵 --- */
    linear-gradient(#918BAD, #918BAD),
    
    /* 
       --- 第二層：2黑鍵群（橫向切齊修正） ---
       我們把 repeating 的方向嚴格限制在 to bottom，
       並確保裡面的線條在 33px 結束後，後面完全是透明碎屑處理。
    */
    repeating-linear-gradient(to bottom, transparent 0px, transparent 11px, #918BAD 11px, #918BAD 18px, transparent 18px, transparent 26px, #918BAD 26px, #918BAD 33px, transparent 33px, transparent 105px),
    
    /* 
       --- 第三層：3黑鍵群（橫向切齊修正） ---
       同樣嚴格卡死 93px 之後的垂直邊界。
    */
    repeating-linear-gradient(to bottom, transparent 0px, transparent 56px, #918BAD 56px, #918BAD 63px, transparent 63px, transparent 71px, #918BAD 71px, #918BAD 78px, transparent 78px, transparent 86px, #918BAD 86px, #918BAD 93px, transparent 93px, transparent 105px),
    
    /* --- 第四層：白鍵交界線（每格精密 15px 無限重複） --- */
    linear-gradient(to bottom, #918BAD 0.75px, transparent 0.75px);

  /* 
     【位置與水平卡死】：
     假設你想讓黑鍵靠左一點，這裡我先幫你設定在 X: 45px 的位置（你可以自己改這個數字）。
     最後一個 0 0 留給白鍵。
  */
  background-position: 16px 12px, 16px 30px, 16px 30px, 0 0;
  
  /* 
     【絕對不准橫向重複】：
     前三層黑鍵全部 no-repeat，最後一層白鍵橫線允許 repeat-y（垂直重複）。
     這樣可以物理阻絕任何想往右邊長出去的靈魂！
  */
  background-repeat: no-repeat, no-repeat, no-repeat, repeat-y;
  
  /* 
     核心尺寸：黑鍵寬度 45px。
     因為 background-repeat: no-repeat 啟動了，
     瀏覽器在畫滿 45px 的黑鍵後，右邊就會直接斷掉，絕對不會有任何線條橫向岔出去！
  */
  background-size: 45px 7px, 45px 735px, 45px 735px, 110px 15px;

  box-shadow: none;
  opacity: 0.3;
  user-select: none;
  pointer-events: none;
  z-index: 2;
}}
```
